1   package fi.ana.gui;
2   
3   import fi.ana.actions.*;
4   import java.awt.Dimension;
5   import javax.swing.*;
6   import fi.ana.logic.Game;
7   import java.awt.BorderLayout;
8   import java.awt.Container;
9   import java.awt.GridLayout;
10  import java.awt.event.KeyEvent;
11  
12  /**
13   *
14   * Creates the UI in which the game runs.
15   */
16  public class GraphicalUI implements Runnable {
17  
18      private Game game;
19      private JFrame frame;
20      private JTextArea hpArea;
21  
22      /**
23       * Constructor.
24       * @param game 
25       */
26      public GraphicalUI(Game game) {
27          this.game = game;
28      }
29  
30      /**
31       * Returns the JFrame object belonging to this class.
32       * @return 
33       */
34      public JFrame getFrame() {
35          return frame;
36      }
37  
38      public void setTextToHpArea(String s) {
39          hpArea.setText(s);
40      }
41  
42      /**
43       * Starts the UI.
44       */
45      @Override
46      public void run() {
47          frame = new JFrame("Game");
48          frame.setPreferredSize(new Dimension(900, 700));
49          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
50          createGameView(frame);
51          createMenu(frame);
52          frame.pack();
53          frame.setVisible(true);
54      }
55  
56      private void createGameView(Container container) {
57          GraphicsArea ga = new GraphicsArea(game);
58          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_W, 0), "up");
59          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_D, 0), "right");
60          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_S, 0), "down");
61          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_A, 0), "left");
62          ga.getActionMap().put("up", new MoveUp(game));
63          ga.getActionMap().put("right", new MoveRight(game));
64          ga.getActionMap().put("down", new MoveDown(game));
65          ga.getActionMap().put("left", new MoveLeft(game));
66          container.add(ga);
67      }
68  
69      private void createMenu(Container container) {
70          JPanel menu = new JPanel(new GridLayout(4, 0));
71          menu.setSize(new Dimension(100, frame.getHeight()));
72          JButton game1 = new JButton("game1");
73          JButton game2 = new JButton("game2");
74          JButton game3 = new JButton("game3");
75          hpArea = new JTextArea("");
76          hpArea.setFont(hpArea.getFont().deriveFont(24f));
77          game1.addActionListener(new Game1Listener(game));
78          game2.addActionListener(new Game2Listener(game));
79          game3.addActionListener(new Game3Listener(game));
80          menu.add(game1);
81          menu.add(game2);
82          menu.add(game3);
83          menu.add(hpArea);
84          container.add(menu, BorderLayout.EAST);
85      }
86  
87      /**
88       * Calls the repaint() method on the UI's JFrame object.
89       */
90      public void refresh() {
91          frame.repaint();
92      }
93  }
