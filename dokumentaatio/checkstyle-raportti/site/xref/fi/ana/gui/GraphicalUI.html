60      @Override
61      public void run() {
62          frame = new JFrame("Game");
63          frame.setPreferredSize(new Dimension(900, 700));
64          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
65          createStartScreen(frame);
66          createMenu(frame);
67          frame.pack();

71      private void createGameView(Container container) {
72          GraphicsArea ga = new GraphicsArea(game);
73          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_W, 0), "up");
74          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_D, 0), "right");
75          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_S, 0), "down");
76          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_A, 0), "left");
77          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_Q, 0), "northwest");
78          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_E, 0), "northeast");
79          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_C, 0), "southeast");
80          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_Z, 0), "southwest");
81          ga.getActionMap().put("up", new MoveUp(game));
82          ga.getActionMap().put("right", new MoveRight(game));
83          ga.getActionMap().put("down", new MoveDown(game));
84          ga.getActionMap().put("left", new MoveLeft(game));
85          ga.getActionMap().put("northwest", new MoveNorthwest(game));
86          ga.getActionMap().put("northeast", new MoveNortheast(game));
87          ga.getActionMap().put("southeast", new MoveSoutheast(game));
88          ga.getActionMap().put("southwest", new MoveSouthwest(game));
89          container.add(ga, 0);
90      }
91  
92      private void createStartScreen(Container container) {
93          StartScreen startScreen = new StartScreen();
94          container.add(startScreen);
95      }
96  
97      private void createGameOverScreen(Container container, int turns) {
98          JPanel gameOver = new JPanel();
99          try {
100             InputStream is = getClass().getClassLoader().getResourceAsStream("youdied.png");
101             BufferedImage image = ImageIO.read(is);
102             ImageIcon icon = new ImageIcon(image);
103             JLabel label = new JLabel(icon);
104             gameOver.add(label);
105             JTextArea teksti = new JTextArea("And survived for " + turns + " turns");
106             teksti.setFont(teksti.getFont().deriveFont(50f));
107             gameOver.add(teksti);
108             container.add(gameOver, 0);
109         } catch (Exception e) {
110             e.printStackTrace();
111         }
112 
113     }
114 
115     private void createMenu(Container container) {
116         JPanel menu = new JPanel(new GridLayout(7, 0));
117         menu.setSize(new Dimension(100, frame.getHeight()));
118         JButton game1 = new JButton("game1");
119         JButton game2 = new JButton("game2");
120         JButton game3 = new JButton("game3");
121         JButton randomGame = new JButton("random");
122         hpArea = new JTextArea("");
123         countdownArea = new JTextArea("");
124         turncountArea = new JTextArea("");
125         hpArea.setFont(hpArea.getFont().deriveFont(15f));
126         countdownArea.setFont(hpArea.getFont().deriveFont(15f));
127         turncountArea.setFont(hpArea.getFont().deriveFont(15f));
128         game1.addActionListener(new Game1Listener(game));
129         game2.addActionListener(new Game2Listener(game));
130         game3.addActionListener(new Game3Listener(game));
131         randomGame.addActionListener(new RandomGameListener(game));
132         menu.add(game1);
133         menu.add(game2);
134         menu.add(game3);
135         menu.add(randomGame);
136         menu.add(hpArea);
137         menu.add(countdownArea);
138         menu.add(turncountArea);
139         container.add(menu, BorderLayout.EAST);
140     }

149     public void startGame() {
150         frame.getContentPane().remove(0);

155         frame.getContentPane().remove(0);
