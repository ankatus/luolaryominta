1   package fi.ana.gui;
2   
3   import fi.ana.actions.*;
4   import java.awt.Dimension;
5   import javax.swing.*;
6   import fi.ana.logic.Game;
7   import java.awt.BorderLayout;
8   import java.awt.Container;
9   import java.awt.GridLayout;
10  import java.awt.event.KeyEvent;
11  
12  public class GraphicalUI implements Runnable {
13  
14      private Game game;
15      private JFrame frame;
16  
17      public GraphicalUI(Game game) {
18          this.game = game;
19      }
20  
21      public JFrame getFrame() {
22          return frame;
23      }
24  
25      @Override
26      public void run() {
27          frame = new JFrame("Game");
28          frame.setPreferredSize(new Dimension(375, 375));
29          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
30          createGameView(frame);
31          createMenu(frame);
32          frame.pack();
33          frame.setVisible(true);
34      }
35  
36      public void createGameView(Container container) {
37          GraphicsArea ga = new GraphicsArea(game);
38          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_W, 0), "up");
39          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_D, 0), "right");
40          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_S, 0), "down");
41          ga.getInputMap(JPanel.WHEN_IN_FOCUSED_WINDOW).put(KeyStroke.getKeyStroke(KeyEvent.VK_A, 0), "left");
42          ga.getActionMap().put("up", new MoveUp(game));
43          ga.getActionMap().put("right", new MoveRight(game));
44          ga.getActionMap().put("down", new MoveDown(game));
45          ga.getActionMap().put("left", new MoveLeft(game));
46          container.add(ga);
47      }
48  
49      public void createMenu(Container container) {
50          JButton restart = new JButton("Restart");
51          restart.addActionListener(new RestartListener(game));
52          container.add(restart, BorderLayout.SOUTH);
53      }
54  
55      public void repaintFrame() {
56          frame.repaint();
57      }
58  
59  }
