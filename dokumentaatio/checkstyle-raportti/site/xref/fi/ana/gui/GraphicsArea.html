1   package fi.ana.gui;

3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import fi.ana.logic.Character;
7   import fi.ana.logic.Game;
8   
9   public class GraphicsArea extends JPanel {
10  
11      Game game;
12  
13      public GraphicsArea(Game game) {
14          super.setBackground(Color.white);
15          this.game = game;
16      }
17  
18      @Override
19      protected void paintComponent(Graphics g) {
20          super.paintComponent(g);
21          drawMap(g);
22      }
23  
24      public void drawMap(Graphics g) {
25          for (int y = 0; y < game.getMap().getSize(); y++) {
26              for (int x = 0; x < game.getMap().getSize(); x++) {
27                  g.setColor(interpretColor(game.getMap().getValue(x, y)));
28                  g.fillRect(x * 15, y * 15, 15, 15);
29              }
30          }
31      }
32  
33      public Color interpretColor(int type) {
34          switch (type) {
35              case 0:
36                  return Color.WHITE;
37              case 1:
38                  return Color.BLACK;
39              case 2:
40                  return Color.RED;
41              case 3:
42                  return Color.BLUE;
43              default:
44                  return null;
45          }
46      }
47  
48  }
