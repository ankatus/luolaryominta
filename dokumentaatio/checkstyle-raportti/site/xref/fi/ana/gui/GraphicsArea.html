1   package fi.ana.gui;
2   
3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import fi.ana.logic.GameCharacter;
7   import fi.ana.logic.Game;
8   import fi.ana.logic.Item;
9   
10  /**
11   *
12   * Draws the graphical representation of the game.
13   */
14  public class GraphicsArea extends JPanel {
15  
16      Game game;
17  
18      /**
19       * Constructor.
20       *
21       * @param game
22       */
23      public GraphicsArea(Game game) {
24          super.setBackground(Color.GRAY);
25          this.game = game;
26      }
27  
28      @Override
29      protected void paintComponent(Graphics g) {
30          super.paintComponent(g);
31          drawMap(g);
32          drawMonstersPlayerAndItems(g);
33      }
34  
35      /**
36       * Draws an appropriately coloured square for each coordinate in the
37       * GameMap.
38       *
39       * @param g
40       */
41      public void drawMap(Graphics g) {
42          for (int y = 0; y < game.getMap().getSize(); y++) {
43              for (int x = 0; x < game.getMap().getSize(); x++) {
44                  if (game.getMap().getValue(x, y)) {
45                      g.setColor(Color.BLACK);
46                      g.fillRect(x * 15, y * 15, 14, 14);
47                  } else {
48                      g.setColor(Color.WHITE);
49                      g.fillRect(x * 15, y * 15, 15, 15);
50                  }
51              }
52          }
53      }
54      
55      public void drawMonstersPlayerAndItems(Graphics g) {
56          g.setColor(Color.RED);
57          for (GameCharacter c : game.getMonsters()) {
58              g.fillRect(c.getX() * 15, c.getY() * 15, 14, 14);
59          }
60          g.setColor(Color.BLUE);
61          g.fillRect(game.getPlayer().getX() * 15, game.getPlayer().getY() * 15, 15, 15);
62          g.setColor(Color.GREEN);
63          for (Item i : game.getItems()) {
64              g.fillRect(i.getX() * 15, i.getY() * 15, 15, 15);
65          }
66      }
67  }
