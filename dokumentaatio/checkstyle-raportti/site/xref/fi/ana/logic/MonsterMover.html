1   package fi.ana.logic;

3   import fi.ana.pathfinding.PathfinderCoordinate;
4   import java.util.Random;
5   import java.util.List;
6   import fi.ana.pathfinding.*;
7   import java.util.ArrayList;
8   
9   /**
10   * Handles monster moving.
11   *
12   */
13  public class MonsterMover {
14  
15      private Game game;
16      private Random r;
17  
18      /**
19       * Constructor.
20       * @param game 
21       */
22      public MonsterMover(Game game) {
23          r = new Random();
24          this.game = game;
25      }
26  
27      /**
28       * Moves all the characters in the input list to random positions.
29       * Characters cannot be moved out of bounds, in walls or on top of other
30       * characters.
31       *
32       * @param monsters characters to be moved.
33       */
34      public void arrangeMonstersRandomly(List<GameCharacter> monsters) {
35          int x;
36          int y;
37          for (GameCharacter c : monsters) {
38              while (true) {
39                  x = r.nextInt(game.getMap().getSize() - 2);
40                  y = r.nextInt(game.getMap().getSize() - 2);
41                  if (!game.checkIfInhabitedCoordinate(x, y) && game.checkIfPassableCoordinate(x, y)) {
42                      game.moveTo(c, x, y);
43                      ArrayList<PathfinderCoordinate> list = new ArrayList();
44                      list.add(new PathfinderCoordinate(c.getX(), c.getY(), null));
45                      c.setPath(new Path(list));
46                      break;
47                  }
48              }
49          }
50      }
51  
52      /**
53       * Moves a character in a random position around its current position.
54       *
55       * @param c character to be moved.
56       */
57      public void moveRandomly(GameCharacter c) {
58          int x;
59          int y;
60          while (true) {
61              x = r.nextInt(3) + c.getX() - 1;
62              y = r.nextInt(3) + c.getY() - 1;
63              if (game.checkIfPassableCoordinate(x, y)) {
64                  if (!game.checkIfInhabitedCoordinate(x, y)) {
65                      game.moveTo(c, x, y);
66                      break;
67                  }
68              }
69          }
70      }
71  
72      /**
73       * Moves a character along it's path.
74       * @param monster character to be moved.
75       */
76      public void moveOnPath(GameCharacter monster) {
77          if (!monster.getPath().getPath().isEmpty()) {
78              int x = monster.getPath().getPathfinderCoordinate(0).getX();
79              int y = monster.getPath().getPathfinderCoordinate(0).getY();
80              if (game.checkIfInhabitedCoordinate(x, y) && game.getPlayer().getX() != x && game.getPlayer().getY() != y) {
81                  return;
82              }
83              game.moveTo(monster, monster.getPath().getPathfinderCoordinate(0).getX(), monster.getPath().getPathfinderCoordinate(0).getY());
84              monster.getPath().remove(0);
85          }
86      }
87  
88      /**
89       * Gets a new path for a character.
90       * @param monster character.
91       */
92      public void getNewPath(GameCharacter monster) {
93          PathfinderCoordinate start = new PathfinderCoordinate(monster.getX(), monster.getY(), null);
94          PathfinderCoordinate end = new PathfinderCoordinate(game.getPlayer().getX(), game.getPlayer().getY(), null);
95          monster.setPath(Pathfinder.findPath(start, end, game.getMap()));
96      }
97  
98      /**
99       * Arranges Item-implementing objects randomly.
100      * @param items items to be arranged.
101      */
102     public void arrangeItemsRandomly(List<Item> items) {
103         int x;
104         int y;
105         for (Item i : items) {
106             while (true) {
107                 x = r.nextInt(game.getMap().getSize() - 2);
108                 y = r.nextInt(game.getMap().getSize() - 2);
109                 if (!game.checkIfInhabitedCoordinate(x, y) && game.checkIfPassableCoordinate(x, y)) {
110                     i.setX(x);
111                     i.setY(y);
112                     break;
113                 }
114             }
115         }
116     }
117 }
