4   import java.util.*;
5   public class MapMaker {
6       
7       
8       public MapMaker() {
9       }

11      public Map makeDefaultMap() {
12          List<Integer> l = new ArrayList<>();
13          l.add(0);
14          l.add(1);
15          Map map = new Map(10, l);
16          populateRow(map, 3, 2, 7, 1);
17          populateColumn(map, 2, 2, 7, 1);
18          return map;
19      }
20      
21      private void populateRow(Map map, int row, int start, int end, int value) {
22          for (int i = start; i <= end; i++) {
23              map.setValue(i, row, value);
24          }
25      }
26      
27      private void populateColumn(Map map, int column, int top, int bottom, int value) {
28          for (int i = top; i <= bottom; i++) {
29              map.setValue(column, i, value);
30          }
31      }
