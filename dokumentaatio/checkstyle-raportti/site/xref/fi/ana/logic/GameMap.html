3   /**
4    * Contains the game's map grid in a two-dimensional array.
5    * Might be changed to contain an array of coordinate-objects instead.
6    */
7   public class GameMap {
8   
9       private int[][] mapValues;
10  
11      public GameMap(int size) {
12          mapValues = new int[size][size];
13      }
14  
15      public int getSize() {
16          return mapValues.length;
17      }
18      
19      /**
20       * Sets the value in the specified position.
21       * If the position is not on the map, does nothing.
22       * @param x
23       * @param y
24       * @param value 
25       */
26      public void setValue(int x, int y, int value) {
27  
28          if (value < 0 || value > 4) {
29              return;
30          } 
31          if (!isValidCoordinate(x, y)) {
32              return;
33          }
34  
35          mapValues[y][x] = value;
36      }
37  
38      /**
39       * Returns the value of the specified position.
40       * If the position is not on the map, returns -1.
41       * @param x
42       * @param y
43       * @return 
44       */
45      public int getValue(int x, int y) {
46          
47          if (!isValidCoordinate(x, y)) {
48              return -1;
49          }
50          
51          return mapValues[y][x];
52      }
53  
54      /**
55       * Checks if the specified position is on the map.
56       * @param x
57       * @param y
58       * @return 
59       */
60      public boolean isValidCoordinate(int x, int y) {
61  
62          int lastIndex = mapValues.length - 1;
63  
64          if (x > lastIndex || x < 0 || y > lastIndex || y < 0) {
65              return false;
66          }
67          
68          return true;
69      }
70      
71      
72      /**
73       * Sets all values to 0.
74       */
75      public void clear() {
76          for (int i = 0; i < mapValues.length; i++) {
77              for (int j = 0; j < mapValues.length; j++) {
78                  mapValues[i][j] = 0;
79              }
80          }
81      }
82  
83  }
